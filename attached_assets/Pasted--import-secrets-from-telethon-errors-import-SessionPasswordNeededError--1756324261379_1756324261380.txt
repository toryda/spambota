# === НИЖЕ ДОБАВИТЬ ===
import secrets
from telethon.errors import SessionPasswordNeededError

# Временные логины в процессе подтверждения: token -> client
_pending_clients: Dict[str, TelegramClient] = {}

async def start_login(api_id: int, api_hash: str, phone: str, proxy_url: Optional[str] = None) -> str:
    """
    Стартуем логин: создаём пустую StringSession, подключаемся,
    отправляем код на телефон, возвращаем токен для продолжения.
    """
    client = TelegramClient(StringSession(), api_id, api_hash, **_proxy_kwargs(proxy_url))
    await client.connect()
    # запросим код
    await client.send_code_request(phone)
    token = secrets.token_urlsafe(16)
    _pending_clients[token] = client
    return token

async def confirm_login(token: str, phone: str, code: str, twofa_password: Optional[str] = None) -> str:
    """
    Завершаем логин: подтверждаем код (и при необходимости 2FA),
    возвращаем итоговый StringSession.
    """
    client = _pending_clients.get(token)
    if not client:
        raise ValueError("Сессия авторизации не найдена или устарела")

    try:
        try:
            await client.sign_in(phone=phone, code=code)
        except SessionPasswordNeededError:
            if not twofa_password:
                # если требуется 2FA, но пароль не дали — просим ещё раз
                raise SessionPasswordNeededError()
            await client.sign_in(password=twofa_password)
    finally:
        # если sign_in бросит любую ошибку — не закрываем клиент, пусть юзер попробует ещё
        ...

    # Готово: достаём строку сессии
    session_str = client.session.save()
    # Уберём из pending: авторизация завершена
    _pending_clients.pop(token, None)
    # НЕ отключаем клиент тут — пусть живёт как активный (его подхватит ensure_client при рассылке)
    return session_str

def cleanup_pending(token: str):
    """На случай отмены/сбоя — закрыть клиент и удалить из pending"""
    client = _pending_clients.pop(token, None)
    if client:
        try:
            # закрыть аккуратно
            client.disconnect()
        except Exception:
            pass
